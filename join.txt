menampilkan semua kolom dari semua tabel untuk setiap pegawai yang ada di tabel utama pegawai
SELECT
    p.*,
    dp.*,
    pk.*
FROM
    pegawai AS p
LEFT JOIN
    detail_pribadi AS dp ON p.nomor_urut_pegawai = dp.nomor_urut_pegawai
LEFT JOIN
    pekerjaan AS pk ON p.nomor_urut_pegawai = pk.nomor_urut_pegawai;

Jika Anda hanya ingin mengambil data untuk satu pegawai tertentu
SELECT
    p.*,
    dp.*,
    pk.*
FROM
    pegawai AS p
LEFT JOIN
    detail_pribadi AS dp ON p.nomor_urut_pegawai = dp.nomor_urut_pegawai
LEFT JOIN
    pekerjaan AS pk ON p.nomor_urut_pegawai = pk.nomor_urut_pegawai
WHERE
    p.nomor_urut_pegawai = 'ID_PEGAWAI_YANG_DICARI'; -- Ganti dengan ID yang sesuai
