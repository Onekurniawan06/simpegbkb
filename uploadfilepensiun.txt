public function showDetailPengajuan($id_pengajuan)
{

    return view('nama_view_anda', compact('pengajuan'));

public function dokumenPersyaratan()
{
    // Sesuaikan 'FilePersyaratanPensiun' dengan nama Model Anda
    // Sesuaikan 'pengajuan_pensiun_id' dengan nama kolom FK di tabel dokumen
    return $this->hasMany(FilePersyaratanPensiun::class, 'pengajuan_pensiun_id', 'id_pengajuan');
}


Route::get('/view-dokumen/{id}', [DokumenController::class, 'viewFile'])->name('dokumen.view');


use App\Models\FilePersyaratanPensiun;
use Illuminate\Support\Facades\Storage;

class DokumenController extends Controller
{
    public function viewFile($id)
    {
        $document = FilePersyaratanPensiun::findOrFail($id);

        // Pastikan file ada di storage sebelum ditampilkan
        if (Storage::disk('local')->exists($document->path_file_server)) {
            return Storage::disk('local')->response($document->path_file_server, $document->nama_file_asli);
        }

        abort(404, 'File not found.');
    }
}
