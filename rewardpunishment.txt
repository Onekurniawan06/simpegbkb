<script>
    function addRewardField() {
        // Ambil elemen kontainer berdasarkan ID
        const container = document.getElementById('reward-container');

        // Buat elemen div baru untuk membungkus grup input
        const newGroup = document.createElement('div');
        newGroup.className = "reward-item p-4 border border-slate-200 rounded-lg bg-slate-50 relative mt-4";

        // Isi konten HTML (sama dengan struktur di atas, namun tombol hapus dimunculkan)
        newGroup.innerHTML = `
            <div class="grid grid-cols-1 md:grid-cols-2 gap-4">
                <div class="flex flex-col gap-1">
                    <label class="text-xs font-bold text-slate-600">Jenis Penghargaan</label>
                    <input type="text" name="jenis_reward[]" placeholder="Contoh: Karyawan Teladan" class="w-full bg-white text-slate-700 text-sm border border-slate-200 rounded-md px-3 py-2 focus:border-blue-500 outline-none shadow-sm" />
                </div>
                <div class="flex flex-col gap-1">
                    <label class="text-xs font-bold text-slate-600">Tanggal Penghargaan</label>
                    <input type="date" name="tanggal_penghargaan[]" class="w-full bg-white text-slate-700 text-sm border border-slate-200 rounded-md px-3 py-2 focus:border-blue-500 outline-none shadow-sm" />
                </div>
                <div class="flex flex-col gap-1">
                    <label class="text-xs font-bold text-slate-600">Diberikan Oleh</label>
                    <input type="text" name="diberikan_oleh[]" placeholder="Nama Instansi/Perusahaan" class="w-full bg-white text-slate-700 text-sm border border-slate-200 rounded-md px-3 py-2 focus:border-blue-500 outline-none shadow-sm" />
                </div>
                <div class="flex flex-col gap-1 md:col-span-2">
                    <label class="text-xs font-bold text-slate-600">Deskripsi/Keterangan</label>
                    <textarea name="deskripsi_reward[]" rows="2" placeholder="Jelaskan detail penghargaan ini..." class="w-full bg-white text-slate-700 text-sm border border-slate-200 rounded-md px-3 py-2 focus:border-blue-500 outline-none shadow-sm"></textarea>
                </div>
            </div>
            <button type="button" onclick="removeRewardField(this)" class="absolute -top-2 -right-2 bg-red-500 text-white rounded-full p-1 hover:bg-red-700 shadow-lg" title="Hapus">
                <svg xmlns="www.w3.org" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12" />
                </svg>
            </button>
        `;

        // Masukkan elemen baru ke dalam kontainer
        container.appendChild(newGroup);
    }

    function removeRewardField(button) {
        // Cari elemen terdekat dengan class .reward-item dan hapus
        const itemToRemove = button.closest('.reward-item');
        itemToRemove.remove();
    }
</script>

                        <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12" />
                    </svg>
                </button>
            </div>
        `;

        container.insertAdjacentHTML('beforeend', punishmentHtml);
    }

    /**
     * Fungsi untuk menghapus grup input punishment tertentu.
     */
    function removePunishmentField(button) {
        // Mencari elemen induk terdekat dengan kelas .punishment-item untuk dihapus
        const group = button.closest('.punishment-item');
        if (group) {
            group.remove();
        }
    }
