<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use Illuminate\Support\Facades\Auth;
use Illuminate\Pagination\LengthAwarePaginator;
use Illuminate\Support\Arr;
use Carbon\Carbon;
use Illuminate\Support\Collection;

class DataPengajuanController extends Controller
{
    // Fungsi utama yang diringkas
    public function formDataPengajuan($type = null)
    {
        $user = Auth::user();

        // 1. Ambil data mentah (sudah with('logs'))
        $allSubmissions = $this->fetchSubmissions($user);

        // Filter berdasarkan type jika ada
        if ($type) {
            $allSubmissions = $allSubmissions->where('type', ucfirst($type));
        }

        $sortedSubmissions = $allSubmissions->sortByDesc('created_at');

        // 2. Proses data (menambahkan processed_flow)
        $processedSubmissions = $this->processSubmissions($sortedSubmissions);

        // 3. Lakukan Pagination
        $paginatedSubmissions = $this->paginateSubmissions($processedSubmissions, 2); // Set perPage ke 2

        // dd($paginatedSubmissions->first()['processed_flow']);

        $pageTitle = 'Data Pengajuan';
        // Anda bisa tambahkan variabel lain jika perlu, misal:
        $breadcrumbs = [
            'Beranda' => route('pegawai.formDataPengajuan'),
            $pageTitle => null
        ];

        // Kirim data yang sudah diolah ke view
        return view('pegawai.datapengajuan', ['submissions' => $paginatedSubmissions])
            ->with(compact('pageTitle', 'breadcrumbs'));
    }

    /**
     * Mengambil data pengajuan mentah dari semua relasi user.
     */
    private function fetchSubmissions($user): Collection
    {
        // Pastikan relasi 'logs' dimuat saat mengambil data dari database
            $cutis = $user->cutis()->with(['logs' => function ($query) {
                $query->orderByDesc('updated_at'); // <-- Pastikan log terbaru ada di paling atas
            }])->get();

            $lemburs = $user->lemburs()->with(['logPersetujuanLembur' => function ($query) {
                $query->orderByDesc('updated_at'); // <-- Terapkan juga untuk lembur
            }])->get();

            $pensiuns = $user->pensiuns()->with(['logPersetujuanPensiun' => function ($query) {
                $query->orderByDesc('updated_at'); // <-- Terapkan juga untuk pensiun
            }])->get();

            // Mapping ulang kunci relasi ke nama standar 'logs' dan konversi ke array
            $processedCuti = $cutis->map(function ($item) {
                $itemArray = $item->toArray();
                $itemArray['type'] = 'Cuti';
                // Pastikan kunci 'logs' ada
                return $itemArray;
            });

            $processedLembur = $lemburs->map(function ($item) {
                $itemArray = $item->toArray();
                $itemArray['type'] = 'Lembur';
                // Ubah nama relasi spesifik 'log_persetujuan_lembur' menjadi 'logs'
                $itemArray['logs'] = $itemArray['log_persetujuan_lembur'] ?? [];
                unset($itemArray['log_persetujuan_lembur']); // Hapus kunci lama
                return $itemArray;
            });

            $processedPensiun = $pensiuns->map(function ($item) {
                $itemArray = $item->toArray();
                $itemArray['type'] = 'Pensiun';
                // Ubah nama relasi spesifik 'log_persetujuan_pensiun' menjadi 'logs'
                $itemArray['logs'] = $itemArray['log_persetujuan_pensiun'] ?? [];
                unset($itemArray['log_persetujuan_pensiun']); // Hapus kunci lama
                return $itemArray;
            });

            return collect()
                ->merge($processedCuti)
                ->merge($processedLembur)
                ->merge($processedPensiun);
        }

    private function processSubmissions(Collection $submissions): Collection
    {
        $flow = ['Pengajuan Awal', 'Manager', 'Direktur Operasional', 'HRO'];
        return $submissions->map(function ($submission) use ($flow) {
            $sortedLogs = collect($submission['logs'] ?? [])->sortByDesc('updated_at');
            $latestLog = $sortedLogs->first();
            $currentStatusFromDB = strtolower($latestLog['status_pengajuan'] ?? 'pending');
            $stageIndex = array_search($latestLog['tahap_persetujuan'] ?? 'Pengajuan Awal', $flow);
            $isOverallRejected = ($currentStatusFromDB == 'ditolak');

            $submission['processed_flow'] = collect($flow)->map(function ($stageName, $index) use ($submission, $stageIndex, $currentStatusFromDB, $isOverallRejected) {
                $isCurrent = ($index == $stageIndex);
                $isCompleted = ($index < $stageIndex) || ($isCurrent && $currentStatusFromDB == 'disetujui');
                $isFailed = ($isOverallRejected && $index <= $stageIndex);

                $themeColor = $isFailed ? 'red' : ($isCompleted ? 'teal' : ($isCurrent ? 'indigo' : 'gray'));

                if ($themeColor === 'red') {
                    $lineColor = 'red';
                } elseif ($themeColor === 'teal') {
                    $lineColor = 'teal';
                } else {
                    $lineColor = 'gray';
                }

                $statusText = ($isOverallRejected && $index > $stageIndex) ? '-' : ($isFailed ? 'Ditolak' : ($isCompleted ? 'Selesai' : ($isCurrent ? 'Proses' : 'Menunggu')));
                $iconType = ($isFailed) ? 'x' : (($isCompleted) ? 'check' : 'clock');
                $stageLog = collect($submission['logs'] ?? [])->firstWhere('tahap_persetujuan', $stageName);

                return [
                    'themeColor' => $themeColor, 'lineColor' => $lineColor, 'statusText' => $statusText,
                    'iconType' => $iconType, 'name' => $stageName, 'isCurrent' => $isCurrent,
                    'updated_at_formatted' => $stageLog ? Carbon::parse($stageLog['updated_at'])->format('d/m/Y H:i') : null,
                ];
            });
            return $submission;
        });
    }

    /**
     * Mengelola logika pagination.
     */
    private function paginateSubmissions(Collection $submissions, int $perPage): LengthAwarePaginator
    {
        $currentPage = LengthAwarePaginator::resolveCurrentPage();
        $currentItems = $submissions->forPage($currentPage, $perPage)->all();

        return new LengthAwarePaginator(
            $currentItems,
            $submissions->count(),
            $perPage,
            $currentPage,
            [
                'path' => request()->url(),
                'query' => request()->query()
            ]
        );
    }
}
