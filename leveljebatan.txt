-- Memperbarui Level Eksekutif/Direksi
UPDATE Jabatan SET level_jabatan = 'Direktur/Eksekutif' WHERE nama_jabatan IN ('Direktur Utama', 'Direktur Kepatuhan', 'Direktur Operasional');

-- Memperbarui Level Manajemen (Senior & Madya)
UPDATE Jabatan SET level_jabatan = 'Manajemen' WHERE nama_jabatan IN ('Kepala SKAI', 'Kepala Satker Kepatuhan & M.R.', 'Manajer Umum', 'Manajer Kredit', 'Manajer Dana & Jasa Layanan', 'Manajer Pelaporan & IT', 'Kepala Kantor Kas');

-- Memperbarui Level Spesialis/Staf Profesional
UPDATE Jabatan SET level_jabatan = 'Staf Profesional/Spesialis' WHERE nama_jabatan IN (
    'Analis Manajemen Risiko, APU-PPT & Strategi Anti Fraud',
    'Auditor Madya',
    'Account Officer',
    'Credit Marketing',
    'Customer Service',
    'Finance Officer',
    'Hardware IT Officer & Programer',
    'Human Resources Officer , Legal, Analis Kepatuhan dan Integritas Pelaporan Keuangan',
    'Treasury Officer'
);

-- Memperbarui Level Operasional/Administrasi
UPDATE Jabatan SET level_jabatan = 'Operasional/Administrasi' WHERE nama_jabatan IN (
    'Administrasi Umum Rumah Tangga / Logistik',
    'Collector Dana',
    'Credit Administration Officer',
    'Credit Maintenance Officer (Remedial)',
    'Funding & Service Banking Officer',
    'Teller'
);
